function searchWord(puzzle, word) {
    const rows = puzzle.length;

    for (let i = 0; i < rows; i++) {
        let gorizontal = puzzle[i].join("");
        if (gorizontal.includes(word) ||
            gorizontal.split("").reverse().join("").includes(word)) {
                return true;
        }
    }
}
